Clasificación de desarrolladoras según ventas totales en Japón:

SELECT d.nombre_desarrolladora, SUM(v.ventas_jp) AS total_ventas_jp
FROM desarrolladora d
INNER JOIN version_videojuego vv ON d.id_desarrolladora = vv.id_desarrolladora
INNER JOIN venta v ON vv.id_version = v.id_version_videojuego
GROUP BY d.nombre_desarrolladora
ORDER BY total_ventas_jp DESC;

Clasificación de desarrolladoras según la suma total de ventas en Japón, Norte América, Unión Europea y otras ventas:

SELECT d.nombre_desarrolladora,
       SUM(v.ventas_jp) AS total_ventas_jp,
       SUM(v.ventas_na) AS total_ventas_na,
       SUM(v.ventas_eu) AS total_ventas_eu,
       SUM(v.otras_ventas) AS total_otras_ventas,
       SUM(v.ventas_jp + v.ventas_na + v.ventas_eu + v.otras_ventas) AS total_ventas_globales
FROM desarrolladora d
INNER JOIN version_videojuego vv ON d.id_desarrolladora = vv.id_desarrolladora
INNER JOIN venta v ON vv.id_version = v.id_version_videojuego
GROUP BY d.nombre_desarrolladora
ORDER BY total_ventas_globales DESC;
